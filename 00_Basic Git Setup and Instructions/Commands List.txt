Create a new repository on the command line
	echo "# IT-Support_Info" >> README.md
	git init
	git add README.md
	git commit -m "first commit"
	git branch -M main
	git remote add origin https://github.com/asnratnakar/IT-Support_Info.git
	git push -u origin main

	git remote add origin https://github.com/asnratnakar/IT-Support_Info.git
	git branch -M main
	git push -u origin main
